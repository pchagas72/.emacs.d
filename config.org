** Theme and Fonts
Configure the colorscheme and typography using ~ewal~ and Spacemacs themes.

#+begin_src emacs-lisp
  (use-package ewal
    :init (setq ewal-use-built-in-always-p nil
                ewal-use-built-in-on-failure-p t
                ewal-built-in-palette "sexy-material"))

  (use-package ewal-spacemacs-themes
    :init (progn
            (setq spacemacs-theme-underline-parens t
                  my:font (font-spec
                           :family "SauceCodePro Nerd Font"
                           :weight 'semi-bold
                           :size 13.0))
            (show-paren-mode +1)
            (global-hl-line-mode)
            (set-frame-font my:font nil t)
            (add-to-list  'default-frame-alist
                          `(font . ,(font-xlfd-name my:font))))
    :config (progn
              (load-theme 'ewal-spacemacs-modern t)
              (enable-theme 'ewal-spacemacs-modern)))

  (use-package ewal-evil-cursors
    :after (ewal-spacemacs-themes)
    :config (ewal-evil-cursors-get-colors
             :apply t :spaceline t))

  (use-package spaceline
    :after (ewal-evil-cursors winum)
    :init (setq powerline-default-separator nil)
    :config (spaceline-spacemacs-theme))

  (defun my/reload-ewal-theme ()
    "Reload ewal colors and re-apply the Spacemacs theme."
    (interactive)
    ;; Check which function your version of ewal uses
    (if (fboundp 'ewal-load-colors)
	(ewal-load-colors)
      (ewal-load-wal-colors))
    
    (load-theme 'ewal-spacemacs-modern t)
    (enable-theme 'ewal-spacemacs-modern)
    (message "ewal theme updated on the fly!"))

  (global-set-key (kbd "C-c r") 'my/reload-ewal-theme)
    #+end_src

* Org Mode
** Visual Enhancements
Make org-mode look like a proper document instead of a raw text file.

#+begin_src emacs-lisp
  (use-package org
    :hook (org-mode . org-indent-mode) ;; Visually indents content under headings
    :custom
    (org-hide-emphasis-markers t)      ;; Hides the *, /, = markers around formatted text
    (org-ellipsis " ▾")                ;; Replaces the folded '...' with a clean arrow
    (org-src-fontify-natively t)       ;; Ensures your code blocks are colorized
    (org-src-tab-acts-natively t)      ;; Makes tab work normally inside code blocks
    (org-startup-folded 'content))     ;; Start with just headings visible

  (use-package org-superstar
    :ensure t
    :after org
    :hook (org-mode . org-superstar-mode)
    :custom
    (org-superstar-headline-bullets-list '("◉" "○" "◈" "◇" "▸"))
    (org-superstar-item-bullet-alist '((?* . ?•)
                                       (?+ . ?▸)
                                       (?- . ?–))))
#+end_src

** Syntax Checking (Flymake)
Real-time error and warning highlighting, seamlessly integrated with Eglot.

#+begin_src emacs-lisp
  (use-package flymake
    :hook (prog-mode . flymake-mode)
    :config
    ;; Optional: slightly tweak how errors are displayed in the minibuffer
    (setq help-at-pt-display-when-idle t)
    
    ;; Evil-mode bindings for quickly jumping between diagnostics
    (with-eval-after-load 'evil
      (define-key evil-normal-state-map (kbd "] d") 'flymake-goto-next-error)
      (define-key evil-normal-state-map (kbd "[ d") 'flymake-goto-prev-error)
      ;; Use leader key or a custom bind to list all errors in a buffer
      (define-key evil-normal-state-map (kbd "C-c d") 'flymake-show-buffer-diagnostics)))
#+end_src
